96  {
97  int r = 0;
98  for (int k = 0 ; k < K ; ++k)
99  {
100  r += snprintf(buffer + r, size - r, "[" INTERVAL_TYPE_MODIFIER ".." INTERVAL_TYPE_MODIFIER "[%c",
101  this->list[k].a, this->list[k].b,
102  (k == K - 1) ? '\0' : '\n');
103  if (r == size)
104  break ;
105  }

108  // return true if intervals intersects on each dimension
109  inline bool
110  intersects(const KHyperrect & intervals) const
111  {
112  for (int k = 0 ; k < K ; ++k)
113  {
114  if (this->list[k].a < intervals.list[k].b && this->list[k].b > intervals.list[k].a)
115  continue ;
116  return false;
117  }
118  return true;
119  }

121  static inline void
122  intersection(
123  KHyperrect * dst,
124  const KHyperrect & x,
125  const KHyperrect & y
126  ) {
127  for (int k = 0 ; k < K ; ++k)
128  {
129  dst->list[k].a = MAX(x.list[k].a, y.list[k].a);
130  dst->list[k].b = MIN(x.list[k].b, y.list[k].b);
131  }
132  }

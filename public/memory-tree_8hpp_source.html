203  * | | |

282  inline void
283  memory_block_init(
284  const Hyperrect & block_rect,
285  const KMemoryBlock & inheriting_block,
286  const Hyperrect & inheriting_rect,
287  const size_t sizeof_type
288  ) {
290  // HOST_VIEW HAS TO BE OFFSET //
292  INTERVAL_DIFF_TYPE_T d[K];
293  Hyperrect::distance_manhattan(inheriting_rect, block_rect, d);
294 
296  // DUPPLICATE REPLICATE INFOS //
298 
299  for (device_global_id_t device_global_id = 0 ; device_global_id < XKRT_DEVICES_MAX ; ++device_global_id)
300  {
301  // retrieve this device replicate
302  MemoryReplicate * replicate = this->replicates + device_global_id;
303  const MemoryReplicate * inheriting_replicate = inheriting_block.replicates + device_global_id;
304 
305  // dupplicate allocations
306  replicate->nallocations = inheriting_replicate->nallocations;
307  for (memory_allocation_view_id_t i = 0 ; i < inheriting_replicate->nallocations ; ++i)
308  {
309  const MemoryReplicateAllocationView * inheriting_allocation = inheriting_replicate->allocations[i];
310 
311  // warning: 'ld' here depends on the allocation itself
312  const INTERVAL_DIFF_TYPE_T offset = d[ACCESS_BLAS_ROW_DIM] + d[ACCESS_BLAS_COL_DIM] * inheriting_allocation->view.ld * sizeof_type;
313  const uintptr_t begin_addr = (uintptr_t) ((INTERVAL_DIFF_TYPE_T) inheriting_allocation->view.addr + offset);
314  assert(begin_addr >= inheriting_allocation->chunk->ptr);
315 
316  # pragma message(TODO "This memory is currently leaked when 'invalidate' is called")
317  MemoryReplicateAllocationView * allocation = new MemoryReplicateAllocationView(inheriting_allocation->chunk, begin_addr, inheriting_allocation->view.ld);
318  replicate->allocations[i] = allocation;
319  // allocation->awaiting must remain empty, tasks will be notified through the shrinked block
320  }
321 
322  // dupplicate fetching / coherency infos
323  replicate->fetching = inheriting_replicate->fetching;
324  replicate->coherency = inheriting_replicate->coherency;
325  }
326 
328  // VALID BITS ARE STILL OK //
330 
331  this->coherency = inheriting_block.coherency;
332  this->fetching = inheriting_block.fetching;
333 
334  # if XKRT_MEMORY_REGISTER_OVERFLOW_PROTECTION
335  this->registered = inheriting_block.registered;
336  # endif /* XKRT_MEMORY_REGISTER_OVERFLOW_PROTECTION */
337  }

795  /* if merging is enabled, merge consecutive transfers to a single transfer */
796  static inline void
797  fetch_list_reduce(fetch_list_t * list)
798  {
799  /* fast way out */
800  const size_t n = list->n;
801  if (n <= 1)
802  return ;
803 
824  /* array of 'n' fetches */
825  fetch_t * fetches = (fetch_t *) (list + 1);
826 
827  /* create vector [0, 1, ..., n-1] */
828  std::vector<int> indices(n);
829  std::iota(indices.begin(), indices.end(), 0);
830 
831  // Sadly, we need to sort here.
832  // Even though data are partially ordered in the KHP tree, the search does not necessarily creates a sorted list.
833  // To remove the 'sort' here, the best way might be to implement a
834  // 'sorted-search' in the KHP-Tree, that search intersecting nodes in-order (left-to-right, bottom-to-top)
835 
836  /* sort the vector so fetches[indices[i]] < fetches[indices[i+1]] in memory */
837  std::sort(
838  std::begin(indices),
839  std::end(indices),
840  [fetches](size_t i, size_t j) {
841  assert(i != j);
842  const fetch_t * fi = fetches + i;
843  const fetch_t * fj = fetches + j;
844  return fi->host_view.begin_addr() < fj->host_view.begin_addr();
845  }
846  );
847 
848  /* find continuous fetches and merge them */
849  fetch_t * fi = fetches + indices[0];
850  assert(fi->merged == false);
851  for (size_t j = 1 ; j < n ; ++j)
852  {
853  fetch_t * fj = fetches + indices[j];
854  assert(fi->host_view.sizeof_type == fj->host_view.sizeof_type);
855 
856  /* already merged */
857  if (fj->merged)
858  continue ;
859 
860  /* fetches must occur between the same devices */
861  if ( fi->src_device_global_id == fj->src_device_global_id &&
862  fi->dst_device_global_id == fj->dst_device_global_id &&
863  fi->dst_chunk == fj->dst_chunk)
864  {
895  const size_t dm = fi->host_view.m;
896  const size_t dn = fj->host_view.n;
897  if (fi->host_view.ld == fj->host_view.ld && (fi->host_view.offset_addr(dm, 0) == fj->host_view.begin_addr() || fi->host_view.offset_addr( 0, dn) == fj->host_view.begin_addr()))
898  {
899  assert(!fi->rects[0].is_empty());
900  assert(!fj->rects[0].is_empty());
901  assert( fi->rects[1].is_empty());
902  assert( fj->rects[1].is_empty());
903 
904  matrix_from_rects(
905  fi->host_view,
906  fi->rects[0], fj->rects[0],
907  fi->host_view.ld,
908  fi->host_view.sizeof_type
909  );
910 
911  fi->rects[1] = fj->rects[0];
912 
913  fj->merged = true;
914  list->fetched();
915  }
916  /* else, try to merge next fetches */
917  else
918  {
919  assert(fj->merged == false);
920  fi = fj;
921  }
922  }
923  }

1365  /* look for a continuous allocation that can store 'access' for the given partition */
1366  inline area_chunk_t *
1367  fetch_access_find_allocation_continuous(
1368  device_global_id_t device_global_id,
1369  Partition & partition
1370  ) {
1371  assert(this->is_locked());
1372 
1373  memory_allocation_view_id_t j = 0;
1374  int nallocations = partition.partites[0].block->replicates[device_global_id].nallocations;
1375  size_t nblocks = partition.partites.size();
1376 
1377  /* for each allocation of the block 0 */
1378  while (j < nallocations)
1379  {
1380  MemoryReplicateAllocationView * rj = partition.partites[0].block->replicates[device_global_id].allocations[j];
1381 
1382  /* for each other blocks */
1383  size_t i = 1;
1384  while (i < nblocks)
1385  {
1386  /* for each allocation of other blocks */
1387  int nallocations = partition.partites[i].block->replicates[device_global_id].nallocations;
1388  for (memory_allocation_view_id_t k = 0 ; k < nallocations ; ++k)
1389  {
1390  /* this block has a view with the same allocation, check next block */
1391  MemoryReplicateAllocationView * rk = partition.partites[i].block->replicates[device_global_id].allocations[k];
1392  if (rj->chunk == rk->chunk)
1393  {
1394  partition.partites[i].dst_allocation_view_id = k;
1395  goto next_block;
1396  }
1397  }
1398 
1399  /* this block has no view view the same allocation, restart from the next view of block 0 */
1400  goto next_view;
1401 
1402 next_block:
1403  ++i;
1404  continue ;
1405  }
1406 
1407  /* every blocks have a view with the allocation 'allocation' */
1408  partition.partites[0].dst_allocation_view_id = j;
1409  return rj->chunk;
1410 
1411 next_view:
1412  ++j;
1413  continue ;
1414  }
1415  return nullptr;
1416  }

1462  inline void
1463  fetch_access_setup_copies(
1464  access_t * access,
1465  device_global_id_t device_global_id,
1466  Partition & partition
1467  ) {
1468  assert(this->is_locked());
1469  assert(access->task);
1470 
1471  // if read mode is set
1472  if (access->mode & ACCESS_MODE_R)
1473  {
1474  const device_global_id_bitfield_t dst_devbit = (device_global_id_bitfield_t) (1 << device_global_id);
1475 
1476  // for each block of that access
1477  for (Partite & partite : partition.partites)
1478  {
1480  // SETUP DST //
1482 
1483  /* destinary allocation view id (cannot be host in this routine) */
1484  const int dst_allocation_view_id = partite.dst_allocation_view_id;
1485  assert(dst_allocation_view_id != MEMORY_REPLICATE_ALLOCATION_VIEW_NONE);
1486  const memory_allocation_view_id_bitfield_t dst_allocbit = (memory_allocation_view_id_bitfield_t) (1 << dst_allocation_view_id);
1487 
1488  /* partite and its view are already coherent on that device */
1489  MemoryReplicate & dst_replicate = partite.block->replicates[device_global_id];
1490  if (dst_replicate.coherency & dst_allocbit)
1491  {
1492  partite.must_fetch = false;
1493  continue ;
1494  }
1495 
1496  /* retrieve an incoherent allocation view */
1497  MemoryReplicateAllocationView * dst_allocation_view = dst_replicate.allocations[dst_allocation_view_id];
1498 
1499  /* increment task fetch counter */
1500  LOGGER_DEBUG(
1501  "task `%s` fetching by `%s` on `%p`",
1502  access->task->label,
1503  (dst_replicate.fetching & dst_allocbit) ? "awaiting" : (partite.block->fetching) ? "forwarding" : "launching",
1504  (void *) dst_allocation_view->view.addr
1505  );
1506 
1507  /* partite is already being fetched on that device */
1508  if (dst_replicate.fetching & dst_allocbit)
1509  {
1510  partite.must_fetch = false;
1511  LOGGER_DEBUG("Skipping fetch of a block already being fetched (concurrent read)");
1512 
1513  /* register a task awaiting on the fetch completion */
1514  __task_fetching(1, access->task);
1515  dst_allocation_view->awaiting.accesses.push_back(access);
1516 
1517  continue ;
1518  }
1519 
1521  // SETUP SRC //
1523 
1524  // find source:
1525  // - if its already coherent on a device, use it as a source
1526  // - else, if its already transfering from the host to any device, wait for it and forward using D2D (PCI contention heuristic)
1527  // - else, transfer H2D
1528 
1529  if (partite.block->coherency & ~(1 << HOST_DEVICE_GLOBAL_ID))
1530  {
1531  partite.must_fetch = true;
1532 
1533  /* create dst view */
1534  partite.dst_device_global_id = device_global_id;
1535  partite.dst_view = dst_allocation_view->view;
1536 
1537  /* get a coherent source */
1538  device_global_id_t src = this->runtime->router.get_source(device_global_id, partite.block->coherency);
1539  assert(partite.block->coherency & (1 << src));
1540 
1541  /* Get the first coherent allocation on that device */
1542  MemoryReplicate & src_replicate = partite.block->replicates[src];
1543  assert(src_replicate.nallocations > 0);
1544  assert(src_replicate.coherency);
1545 
1546  # pragma message(TODO "Instead of getting the first coherent, maybe get the LARGEST coherent, so we maximize odds to merge with future allocations")
1547  memory_allocation_view_id_t src_allocation_view_id = (memory_allocation_view_id_t) (__builtin_ffs(src_replicate.coherency) - 1);
1548  assert(src_replicate.coherency & (1 << src_allocation_view_id));
1549  assert(0 <= src_allocation_view_id && src_allocation_view_id < src_replicate.nallocations);
1550 
1551  /* Retrieve and set src view infos */
1552  MemoryReplicateAllocationView * src_allocation_view = src_replicate.allocations[src_allocation_view_id];
1553  partite.src_device_global_id = src;
1554  partite.src_allocation_view_id = src_allocation_view_id;
1555  partite.src_view = src_allocation_view->view;
1556  partite.src_chunk = src_allocation_view->chunk;
1557  }
1558  # if USE_D2D_FORWARDING
1559  /* heuristic: if another device is already fetching from the host, register a forward callback instead to reduce PCI contention */
1560  else if (partite.block->fetching & ~(1 << HOST_DEVICE_GLOBAL_ID))
1561  {
1562  /* the fetch will be initiated by the other device that
1563  * is already fetching that data for a D2D transfer.
1564  * No need to create partite.src and partite.dst as we
1565  * are not fetching now */
1566  partite.must_fetch = false;
1567 
1568  /* one device is already fetching, add a D2D forward callback */
1569  device_global_id_t fetching_device_global_id = this->runtime->router.get_source(device_global_id, partite.block->fetching);
1570  assert(0 <= fetching_device_global_id && fetching_device_global_id < XKRT_DEVICES_MAX);
1571  assert(partite.block->fetching & (1 << fetching_device_global_id));
1572 
1573  MemoryReplicate & fetching_replicate = partite.block->replicates[fetching_device_global_id];
1574  assert(fetching_replicate.fetching);
1575 
1576  // Maybe there is several fetching alloc ? in such case, which one to pick ?
1577  // Currently select the first one
1578  memory_allocation_view_id_t fetching_allocation_view_id = (memory_allocation_view_id_t) (__builtin_ffs(fetching_replicate.fetching) - 1);
1579  assert(0 <= fetching_allocation_view_id && fetching_allocation_view_id < fetching_replicate.nallocations);
1580  assert(fetching_replicate.fetching & (1 << fetching_allocation_view_id));
1581 
1582  MemoryReplicateAllocationView * fetching_allocation_view = fetching_replicate.allocations[fetching_allocation_view_id];
1583  assert(fetching_allocation_view);
1584 
1585  LOGGER_DEBUG("registered a forward for task `%s`", access->task->label);
1586 
1587  const MemoryForward forward(access, partition.chunk, partite.hyperrect, device_global_id, dst_allocation_view->view);
1588  fetching_allocation_view->awaiting.forwards.push_back(forward);
1589  __task_fetching(1, access->task);
1590  }
1591  # endif /* USE_D2D_FORWARDING */
1592  else
1593  {
1594  assert(!partite.block->coherency);
1595  # if USE_D2D_FORWARDING
1596  assert(!partite.block->fetching);
1597  # endif /* !USE_D2D_FORWARDING */
1598 
1599  partite.must_fetch = true;
1600 
1601  /* create dst view */
1602  partite.dst_device_global_id = device_global_id;
1603  partite.dst_view = dst_allocation_view->view;
1604 
1605  /* using host as src, which is assumed coherent */
1606  partite.src_device_global_id = HOST_DEVICE_GLOBAL_ID;
1607  partite.src_allocation_view_id = MEMORY_REPLICATE_ALLOCATION_VIEW_NONE;
1608  }
1609 
1610  /* update bitfields so no other concurrent fetch occurs */
1611  dst_replicate.fetching |= dst_allocbit;
1612  partite.block->fetching |= dst_devbit;
1613 
1615  // ASSERTION ON SRC AND DST //
1617 
1618  assert(partite.dst_device_global_id != partite.src_device_global_id ||
1619  partite.dst_allocation_view_id != partite.src_allocation_view_id);
1620  }
1621  }
1622  }

1938  /* shrinking on dimension 'k' from 'this->hyperrect[k]' to 'interval' */
1939  void
1940  on_shrink(
1941  NodeBase * nodebase,
1942  const Interval & interval,
1943  int k
1944  ) {
1945  static_assert(K == 2);
1946  Node * node = reinterpret_cast<Node *>(nodebase);
1947 
1948  assert(k < K);
1949  assert(node->hyperrect[k].includes(interval));
1950 
1952  // SHRINK HOST VIEW //
1954 
1955  assert(node->hyperrect[k].a <= interval.a);
1956  const INTERVAL_DIFF_TYPE_T da = interval.a - node->hyperrect[k].a;
1957 
1958  assert(node->hyperrect[k].b >= interval.b);
1959 
1960  // must be aligned on sizeof(type)
1961  if (k == ACCESS_BLAS_ROW_DIM)
1962  {
1963  const INTERVAL_DIFF_TYPE_T db = node->hyperrect[k].b - interval.b;
1964  (void) db;
1965  assert(da % this->sizeof_type == 0);
1966  assert(db % this->sizeof_type == 0);
1967  }
1968 
1969  // shrinked-left, gotta offset the views

50  if (valid & (1 << dst))

52 
53  /* find a device for P2P transfer - lowest rank <=> best performance */
54  for (int rank = 0 ; rank < XKRT_DEVICES_PERF_RANK_MAX - 1 ; ++rank)
55  {
56  /* get valid devices for this perf */
57  const device_global_id_bitfield_t mask = valid & this->affinity[dst][rank];
58  if (mask == 0)
59  continue ;
60 
61  /* return a random device with this affinity */
62  return __random_set_bit(mask) - 1;
63  }
64 
65  /* get any random device */
66  return (device_global_id_t) (__random_set_bit(valid) - 1);

80 # define LOGGER_PRINT_LINE() \
81  fprintf(LOGGER_FD, "%s:%d (%s)\n", __FILE__, __LINE__, __func__);
82 
83 # define LOGGER_PRINT(LVL, ...) \
84  do { \
85  if (LVL <= LOGGER_VERBOSE) \
86  { \
87  SPINLOCK_LOCK(LOGGER_PRINT_MTX); \
88  struct timespec _ts; \
89  clock_gettime(CLOCK_MONOTONIC, &_ts); \
90  uint64_t t = (uint64_t)(_ts.tv_sec * 1000000000) + \
91  (uint64_t) _ts.tv_nsec; \
92  if (LOGGER_LAST_TIME != 0) \
93  LOGGER_TIME_ELAPSED += (double) (t - LOGGER_LAST_TIME) / 1e9; \
94  LOGGER_LAST_TIME = t; \
95  if (isatty(STDOUT_FILENO)) \
96  fprintf(LOGGER_FD, "[%8lf] " \
97  "[TID=%d] " \
98  "[\033[1;37m" LOGGER_HEADER "\033[0m] " \
99  "[%s%s\033[0m] ", \
100  LOGGER_TIME_ELAPSED, \
101  gettid(), \
102  LOGGER_PRINT_COLORS[LVL], \
103  LOGGER_PRINT_HEADERS[LVL]); \
104  else \
105  fprintf(LOGGER_FD, "[%8lf]" \
106  "[TID=%d] " \
107  "[" LOGGER_HEADER "] " \
108  "[%s] ", \
109  LOGGER_TIME_ELAPSED, \
110  gettid(), \
111  LOGGER_PRINT_HEADERS[LVL]); \
112  fprintf(LOGGER_FD, __VA_ARGS__); \
113  fprintf(LOGGER_FD, "\n"); \
114  fflush(LOGGER_FD); \
115  SPINLOCK_UNLOCK(LOGGER_PRINT_MTX); \
116  } \
117  if (LVL == LOGGER_PRINT_FATAL_ID) \
118  { \
119  LOGGER_PRINT_LINE(); \
120  fflush(LOGGER_FD); \
121  abort(); \
122  } \
123  } while (0)

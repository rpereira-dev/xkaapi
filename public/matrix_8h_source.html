148  inline uintptr_t
149  offset_addr(const size_t offset_m, const size_t offset_n) const
150  {
151  assert(this->storage == MATRIX_ROWMAJOR || this->storage == MATRIX_COLMAJOR);
152  assert(offset_n >= 0);
153  assert(offset_m >= 0);
154 
155  switch (this->storage)
156  {
157  case (MATRIX_ROWMAJOR):
158  return this->addr + ((size_t)offset_n * this->sizeof_type) +
159  ((size_t)offset_m * this->sizeof_type * this->ld);
160 
161  case (MATRIX_COLMAJOR):
162  return this->addr + ((size_t)offset_n * this->sizeof_type * this->ld) +
163  ((size_t)offset_m * this->sizeof_type);
164  default:
165  abort();
166  }
167  }
168 
169  /* return end address */
170  inline uintptr_t
171  end_addr(void) const
172  {
173  assert(this->storage == MATRIX_ROWMAJOR || this->storage == MATRIX_COLMAJOR);
174  switch (this->storage)
175  {
176  case (MATRIX_ROWMAJOR):
177  return this->addr +
178  (this->n * this->sizeof_type) +
179  (this->m * this->sizeof_type * this->ld);
180 
181  case (MATRIX_COLMAJOR):
182  return this->addr +
183  (this->n * this->sizeof_type * this->ld) +
184  (this->m * this->sizeof_type);
185 
186  default:
187  abort();
188  }
189  }

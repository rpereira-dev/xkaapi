41 typedef enum access_mode_t : int
42 {
43  ACCESS_MODE_R = 0b00000001, // read
44  ACCESS_MODE_W = 0b00000010, // write
45  ACCESS_MODE_RW = ACCESS_MODE_R | ACCESS_MODE_W,
46  ACCESS_MODE_V = 0b00000100, // incoherent access (= 'virtual' = dont really move the memory)
47  ACCESS_MODE_VW = ACCESS_MODE_W | ACCESS_MODE_V,
48  ACCESS_MODE_D = 0b00001000, // detached access = do not fulfill
49  // dependencies on task completion: the
50  // task is responsible of fulfillment
51  // the access itself
52 } access_mode_t;
53 
54 inline access_mode_t &
55 operator|=(access_mode_t & lhs, access_mode_t rhs) {
56  lhs = static_cast<access_mode_t>(
57  static_cast<int>(lhs) | static_cast<int>(rhs)
58  );
59  return lhs;
60 }

87 static inline const char *
88 access_mode_to_str(access_mode_t mode)
89 {
90  switch (mode)
91  {
92  case (ACCESS_MODE_V): return "ACCESS_MODE_V";
93  case (ACCESS_MODE_R): return "ACCESS_MODE_R";
94  case (ACCESS_MODE_W): return "ACCESS_MODE_W";
95  case (ACCESS_MODE_RW): return "ACCESS_MODE_RW";
96  default: return "ACCESS_MODE_UNKN";
97  }
98 }
